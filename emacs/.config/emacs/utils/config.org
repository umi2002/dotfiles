#+begin_src emacs-lisp
  (defun toggle_vterm ()
    "Toggle between vterm and current buffer."
    (interactive)
    (if (string= major-mode "vterm-mode")
	(evil-switch-to-windows-last-buffer)
      (if (get-buffer "vterm")
	  (switch-to-buffer "vterm")
	(vterm))))
